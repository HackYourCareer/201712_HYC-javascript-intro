Function Parameters Handling



function sum(a, b, c, d) {
  return a + b + c + d;
}

let result = sum(1, 2); // NaN


function sum(a, b, c, d) {
  if (c === undefined) {
    c = 0;
  }
  if (d === undefined) {
    d = 0;
  }

  return a + b + c + d;
}

let result = sum(1, 2); // 3

Function Parameters Handling



function sum(a, b, c = 0, d = 0) {
  return a + b + c + d;
}

let result = sum(1, 2); // 3
